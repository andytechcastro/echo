func (s *SQLiteVecStore) VectorSearch(ctx context.Context, queryVec []float32, project string, limit int) ([]domain.SearchResult, error) {
    query := `
        SELECT v.learning_id, v.distance, l.*
        FROM vec_learnings v
        JOIN learnings_data l ON l.id = v.learning_id
        WHERE v.embedding MATCH ?
          AND v.project = ?

Search flow in usecase:
```go
func (s *SearchLearning) Execute(ctx context.Context, query *domain.SearchQuery) ([]domain.SearchResult, error) {
    if s.embedder != nil {
        queryVec, err := s.embedder.Embed(ctx, query.Query)
        if err == nil {
            results, err := s.store.VectorSearch(ctx, queryVec, query.Project, query.Limit)
            if err == nil && len(results) > 0 {
                return results, nil
            }
        }
        slog.Warn("vector search failed, falling back to BM25", "error", err)
    }
    return s.store.Search(ctx, query)
}
```
